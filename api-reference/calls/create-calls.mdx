## Request Body

<ParamField body="calls" type="array" required>
  Array of call objects
  
  <Expandable title="Call Object">
    <ParamField body="externalCallId" type="string" required>
      External unique ID from call provider
    </ParamField>
    
    <ParamField body="direction" type="string" default="unknown">
      Call direction. Options: `inbound`, `outbound`, `unknown`
    </ParamField>
    
    <ParamField body="status" type="string" default="completed">
      Call status. Options: `completed`, `missed`, `voicemail`, `failed`, `busy`, `cancelled`, `unknown`
    </ParamField>
    
    <ParamField body="fromNumber" type="string">
      Phone number the call originated from
    </ParamField>
    
    <ParamField body="toNumber" type="string">
      Phone number the call was made to
    </ParamField>
    
    <ParamField body="durationSeconds" type="number">
      Call duration in seconds
    </ParamField>
    
    <ParamField body="ringDurationSeconds" type="number">
      Ring duration before answer
    </ParamField>
    
    <ParamField body="startTimestamp" type="string">
      Call start time (ISO 8601)
    </ParamField>
    
    <ParamField body="endTimestamp" type="string">
      Call end time (ISO 8601)
    </ParamField>
    
    <ParamField body="recordingUrl" type="string">
      URL to call recording
    </ParamField>
    
    <ParamField body="customer" type="object">
      Customer info with `phone`, `email`, and `name` fields
    </ParamField>
    
    <ParamField body="orgUnitIds" type="array">
      Array of org unit UUIDs this call belongs to
    </ParamField>
    
    <ParamField body="orgUnitId" type="string">
      Single org unit UUID (alternative to orgUnitIds)
    </ParamField>
    
    <ParamField body="metadata" type="object">
      Additional metadata
    </ParamField>
    
    <ParamField body="title" type="string">
      Call title (auto-generated if not provided)
    </ParamField>
  </Expandable>
</ParamField>
