---
title: 'Create Companies'
api: 'POST /api/v1/companies'
description: 'Batch create or update company records'
---

## Request Body

<ParamField body="companies" type="array" required>
  Array of company objects to create or update
  
  <Expandable title="Company Object">
    <ParamField body="name" type="string" required>
      Company display name
    </ParamField>
    
    <ParamField body="externalId" type="string" required>
      **Primary External ID**. Used for integration mapping lookup.
    </ParamField>
    
    <ParamField body="domain" type="string">
      Company website domain
    </ParamField>
    
    <ParamField body="externalCompanyId" type="string">
      **Secondary External ID**. Stored on the record. Defaults to `externalId` if not provided.
    </ParamField>

    <Info>
      **Upsert Logic:**
      1. **Mapping Lookup**: Checks for an existing mapping using `externalId`. If found, updates the linked company.
      2. **DB Constraint**: If no mapping is found, attempts to insert. If a company with the same `externalCompanyId` exists, it updates that record (and creates a new mapping).
      3. **Insert**: If neither is found, a new company is created.
    </Info>
    
    <ParamField body="stageKey" type="string" default="lead">
      Company stage key. Options: `lead`, `prospect`, `customer`, `expansion`, `churned`, `lost`, `partner`, `other`
    </ParamField>
    
    <ParamField body="ownerUserId" type="string">
      Owner user UUID
    </ParamField>
    
    <ParamField body="healthScore" type="integer">
      Health score (0-100)
    </ParamField>
    
    <ParamField body="arr" type="integer">
      Annual Recurring Revenue
    </ParamField>
    
    <ParamField body="attributes" type="object">
      Custom key-value attributes
    </ParamField>
    
    <ParamField body="properties" type="object">
      Custom field values (maps to field_values table)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Whether the operation was successful
</ResponseField>

<ResponseField name="summary" type="object">
  <Expandable title="Summary Object">
    <ResponseField name="total" type="integer">
      Total number of companies processed
    </ResponseField>
    <ResponseField name="success" type="integer">
      Number of companies successfully created/updated
    </ResponseField>
    <ResponseField name="failed" type="integer">
      Number of companies that failed
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="results" type="array">
  Array of result objects for each company
</ResponseField>
