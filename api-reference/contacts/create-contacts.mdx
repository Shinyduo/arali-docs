---
title: 'Create Contacts'
api: 'POST /api/v1/contacts'
description: 'Batch create or update contact records'
---

## Request Body

<ParamField body="contacts" type="array" required>
  Array of contact objects to create or update
  
  <Expandable title="Contact Object">
    <ParamField body="externalId" type="string" required>
      External unique ID from integration source
    </ParamField>
    
    <ParamField body="fullName" type="string">
      Contact's full name
    </ParamField>
    
    <ParamField body="primaryEmail" type="string">
      Primary email address
    </ParamField>
    
    <ParamField body="primaryPhone" type="string">
      Primary phone number
    </ParamField>
    
    <ParamField body="title" type="string">
      Job title
    </ParamField>
    
    <ParamField body="ownerUserId" type="string">
      Owner user UUID
    </ParamField>
    
    <ParamField body="emails" type="array">
      List of email objects with `email`, `isPrimary`, and `label` fields
    </ParamField>
    
    <ParamField body="phones" type="array">
      List of phone objects with `phone`, `isPrimary`, and `label` fields
    </ParamField>
    
    <ParamField body="companies" type="array">
      List of company association objects with `companyId`, `relation`, `isPrimary`, and `attributes` fields
    </ParamField>
    
    <ParamField body="attributes" type="object">
      Custom key-value attributes
    </ParamField>
    
    <ParamField body="properties" type="object">
      Custom field values
    </ParamField>
  </Expandable>
</ParamField>
