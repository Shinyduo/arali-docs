---
title: 'Batch Set Field Values'
api: 'POST /api/v1/fields/values/batch'
description: 'Set field values for multiple entities'
---

## Request Body

<ParamField body="entityType" type="string" required>
  Entity type. Options: `company`, `account`, `contact`
</ParamField>

<ParamField body="entities" type="array" required>
  Array of entity field value objects
  
  <Expandable title="Entity Object">
    <ParamField body="entityId" type="string" required>
      Entity UUID
    </ParamField>
    
    <ParamField body="fields" type="array" required>
      Array of field value objects with `fieldKey` and `value`
    </ParamField>
  </Expandable>
</ParamField>
