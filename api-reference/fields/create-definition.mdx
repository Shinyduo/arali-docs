---
title: 'Create Field Definition'
api: 'POST /api/v1/fields/definitions'
description: 'Create a new custom field definition'
---

## Request Body

<ParamField body="fieldKey" type="string" required>
  Unique key for the field (snake_case recommended)
</ParamField>

<ParamField body="fieldName" type="string" required>
  Display name for the field
</ParamField>

<ParamField body="fieldType" type="string" required>
  Field type. Options: `text`, `number`, `date`, `boolean`, `json`, `enum`
</ParamField>

<ParamField body="entityType" type="string" required>
  Entity type this field applies to. Options: `company`, `account`, `contact`
</ParamField>

<ParamField body="enumOptions" type="array">
  Valid options for enum field type (required if fieldType is `enum`)
</ParamField>

<ParamField body="isRequired" type="boolean" default="false">
  Is this field required
</ParamField>

<ParamField body="displayOrder" type="integer">
  Order for UI display
</ParamField>
