## Request Body

<ParamField body="entityType" type="string" required>
  Entity type. Options: `company`, `account`, `contact`
</ParamField>

<ParamField body="entityId" type="string" required>
  Entity UUID
</ParamField>

<ParamField body="fields" type="array" required>
  Array of field value objects
  
  <Expandable title="Field Value Object">
    <ParamField body="fieldKey" type="string" required>
      Key of the field definition
    </ParamField>
    
    <ParamField body="value" type="any" required>
      Value matching the field type
    </ParamField>
  </Expandable>
</ParamField>
