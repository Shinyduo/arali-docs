---
title: 'Create Meeting'
api: 'POST /api/v1/meetings'
description: 'Create or update a meeting record'
---

## Request Body

<ParamField body="scheduledStartAt" type="string" required>
  Scheduled start time (ISO 8601)
</ParamField>

<ParamField body="scheduledEndAt" type="string" required>
  Scheduled end time (ISO 8601). Must be after start and in the future.
</ParamField>

<ParamField body="title" type="string" default="empty title">
  Meeting title/subject
</ParamField>

<ParamField body="channel" type="string">
  Meeting channel. Options: `gmeet`, `zoom`, `teams`, `offline`
</ParamField>

<ParamField body="joinUrl" type="string">
  URL to join the meeting
</ParamField>

<ParamField body="status" type="string" default="scheduled">
  Meeting status. Options: `scheduled`, `skip`, `in_progress`, `cancelled`, `completed`, `deleted`, `error`
</ParamField>

<ParamField body="externalMeetingId" type="string">
  External unique ID from calendar source (used for upsert)
</ParamField>

<ParamField body="hostEmail" type="string">
  Email of meeting host/organizer
</ParamField>

<ParamField body="participantEmails" type="array">
  Array of participant email addresses
</ParamField>

<ParamField body="orgUnitIds" type="array">
  Array of org unit UUIDs this meeting belongs to
</ParamField>

<ParamField body="orgUnitId" type="string">
  Single org unit UUID (alternative to orgUnitIds)
</ParamField>

<ParamField body="recurringMeetingId" type="string">
  Recurring meeting reference ID
</ParamField>

<ParamField body="type" type="string">
  Meeting type. Options: `external`, `internal` (auto-determined from participant domains if not provided)
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata
</ParamField>
